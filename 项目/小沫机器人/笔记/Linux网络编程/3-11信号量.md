# 1.用法

## 1.1 信号量初始化

```c
#include <semaphore.h>
sem_t
int sem_init(sem_t *sem, int pshared, unsigned int value);
/*
    - 功能：初始化一个信号量
    - sem：要初始化的信号量
    - pshared：共享方式
        - 0：线程间共享（同一进程中的线程）
        - 非 0：进程间共享
    - value：信号量的初始值
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

## 1.2 销毁信号量

```c
int sem_destroy(sem_t *sem);
/*
    - 功能：销毁一个信号量
    - sem：要销毁的信号量 
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

- 注意：销毁前，不能还有线程/进程正在使用该信号量

## 1.3 申请一个信号量资源

### 1.3.1 阻塞方式

```c
int sem_wait(sem_t *sem);
/*
    - 功能：申请一个信号量资源（P 操作）
    - sem：要操作的信号量
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

- 说明：
    - 如果信号量值大于 0，则减 1 后立即返回
    - 如果信号量值等于 0，则阻塞等待

### 1.3.2 非阻塞方式

```c
int sem_trywait(sem_t *sem);
/*
    - 功能：以非阻塞方式申请一个信号量资源
    - sem：要操作的信号量
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

- 说明：
    - 如果信号量值大于 0，则减 1 后立即返回
    - 如果信号量值等于 0，则立即失败，不阻塞

### 1.3.3 指定时间内申请一个信号量资源

```c
int sem_timedwait(sem_t *sem, const struct timespec *abs_timeout);
/*
    - 功能：在指定时间内申请一个信号量资源
    - sem：要操作的信号量
    - abs_timeout：绝对超时时间
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

- 说明：
    - 如果信号量值大于 0，则减 1 后立即返回
    - 如果信号量值等于 0，则等待直到超时或获取成功
    - 超时会返回失败，`errno：ETIMEDOUT`

## 1.4 释放信号量

```c
int sem_post(sem_t *sem);
/*
    - 功能：释放一个信号量资源（V 操作）
    - sem：要操作的信号量
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

- 说明：
    - 将信号量值加 1
    - 如果有线程/进程正在等待该信号，会唤醒其中一个

## 1.5 获取信号量的值

```c
int sem_getvalue(sem_t *sem, int *sval);
/*
    - 功能：获取信号量当前的值
    - sem：要操作的信号量
    - sval：用于保存信号量当前值的变量地址
    - 返回值：
        - 成功：0
        - 失败：-1，并设置 errno
*/
```

- 说明：
    - 读取的是当前时刻的值
    - 多线程/多进程环境下，该值可能马上发生变化

# 2.案例

使用信号量对3-10的例子进行改进：

```c
#include <stdio.h>
#include <pthread.h>
#include <malloc.h>
#include <unistd.h>
#include <semaphore.h>

typedef struct _Node
{
    int val;
    struct _Node *next;
} Node;

static Node *head = NULL, *tail = NULL;
pthread_mutex_t mutex;
sem_t semaphore;

Node *push(int val)
{
    if (head == NULL)
    {
        head = (Node *)malloc(sizeof(Node));
        head->val = val;
        head->next = NULL;
        tail = head;
    }
    else
    {
        tail->next = (Node *)malloc(sizeof(Node));
        tail = tail->next;
        tail->val = val;
        tail->next = NULL;
    }

    return tail;
}

int pop()
{
    if (head != NULL)
    {
        Node *tmp = head;
        head = head->next;
        int val = tmp->val;
        free(tmp);
        return val;
    }
    else
    {
        return -1;
    }
}

void traverse()
{
    Node *ptr = head;
    while (ptr)
    {
        printf("%d->", ptr->val);
        ptr = ptr->next;
    }
    printf("EOF\n");
}

void *consume(void *arg)
{
    while (1)
    {
        sem_wait(&semaphore);
        pthread_mutex_lock(&mutex);
        if (pop() < 0)
        {
            printf("queue empty, pop failed.\n");
        }
        else
        {
            printf("consumer pop, current queue: ");
            traverse();
        }
        pthread_mutex_unlock(&mutex);
        sleep(1);
    }
    return NULL;
}

void *produce(void *arg)
{
    static int num = 0;
    while (1)
    {
        pthread_mutex_lock(&mutex);
        push(num++);
        printf("producer push, current queue: ");
        traverse();
        pthread_mutex_unlock(&mutex);
        sem_post(&semaphore); // 释放一个信号量
        sleep(5);
    }
    return NULL;
}

int main(void)
{
    pthread_mutex_init(&mutex, NULL);
    sem_init(&semaphore, 0, 0);

    pthread_t consumer1, prod1, prod2, prod3;
    pthread_create(&consumer1, NULL, consume, NULL);
    pthread_create(&prod1, NULL, produce, NULL);
    pthread_create(&prod2, NULL, produce, NULL);
    pthread_create(&prod3, NULL, produce, NULL);

    pthread_detach(prod3);
    pthread_detach(prod2);
    pthread_detach(prod1);
    pthread_detach(consumer1);

    while (1)
    {
        sleep(10);
    }

    free(head);
    sem_destroy(&semaphore);
    pthread_mutex_destroy(&mutex);
}
```
