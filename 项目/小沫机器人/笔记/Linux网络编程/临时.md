信号捕获函数

或者可以称为信号处理函数。当信号来临的时候，不按照默认行为，而是执行信号捕获函数。

signal（基础版）

```c
#include <signal.h>

void (*signal(int sig, void (*handler)(int)))(int);
// void (* ... )(int)：返回一个指向void func(int)的函数指针
```

下面是案例：

```c
#include <stdio.h>
#include <signal.h>

void handler(int sig) {
    printf("收到信号: %d\n", sig);
}

int main() {
    signal(SIGINT, handler);  // 捕获 Ctrl+C

    while (1);
}
```

signaction（标准版）

```c
#include <signal.h>

int sigaction(
    int signum,
    const struct sigaction *act,	// -I，本次要设置的处理方式
    struct sigaction *oldact		// -O，原来设置的处理方式
);
```

- `struct sigaction`：

  ```c
  struct sigaction {
      void (*sa_handler)(int);	// 简单处理函数
      void (*sa_sigaction)(int, siginfo_t *, void *); // 高级处理函数，输入包含信号的基本信息
      sigset_t sa_mask;	// 屏蔽信号：在执行信号处理函数期间，临时“屏蔽某些信号”
      int sa_flags;	// 标志位
  };
  ```

  - `siginfo_t`是一个结构体类型，成员包含：

    ```c
    #include <signal.h>
    
    /*
     * siginfo_t：描述“信号的详细信息”
     * 注意：不同信号可用的字段不同（不是全部字段都有意义）
     */
    
    typedef struct siginfo {
        int si_signo;   // 信号编号（如 SIGINT、SIGUSR1）
        int si_errno;   // 错误码（一般很少用）
        int si_code;    // 信号来源类型（非常重要）
    
        /*
         * si_code 常见取值：
         * SI_USER   → 用户通过 kill 发送
         * SI_KERNEL → 内核发送
         * SI_QUEUE  → 通过 sigqueue 发送（可携带数据）
         */
    
        // ===== 发送者信息 =====
        pid_t si_pid;   // 发送信号的进程ID
        uid_t si_uid;   // 发送信号的用户ID
    
        // ===== 携带的数据（实时信号）=====
        union sigval {
            int   sival_int;   // 发送的整数数据
            void *sival_ptr;   // 发送的指针数据
        } si_value;
    
        /*
         * 只有使用 sigqueue() 发送信号时才有效：
         *
         * sigqueue(pid, SIGUSR1, val);
         */
    
        // ===== 子进程相关 =====
        int si_status;  // 子进程退出状态（SIGCHLD时用）
        clock_t si_utime; // 子进程用户态CPU时间
        clock_t si_stime; // 子进程内核态CPU时间
    
        // ===== 内存错误相关 =====
        void *si_addr;  // 出错地址（如 SIGSEGV 段错误）
    
    } siginfo_t;
    ```

  - `sa_flags`，标志位，常见标志如下：

    | 标志                              | 作用                                                         |
    | :-------------------------------- | :----------------------------------------------------------- |
    | `SA_RESTART`                      | 使被该信号中断的某些系统调用（如 `read`, `write`, `open` 等）自动重启，而不是返回 `EINTR` 错误。 |
    | `SA_NODEFER`（或 `SA_NOMASK`）    | 在处理信号函数执行期间，**不阻塞当前正在处理的信号本身**。默认情况下，执行处理函数时会自动阻塞当前信号，防止重入。 |
    | `SA_RESETHAND`（或 `SA_ONESHOT`） | 在处理函数执行一次后，将该信号的处理方式恢复为默认（通常是终止进程或忽略）。 |
    | `SA_SIGINFO`                      | 使用 `sa_sigaction` 替代 `sa_handler` 作为处理函数，可获取更多信号信息（如发送者 pid、用户态上下文等）。 |

## 信号集

信号集本质就是一个“位图（bitmask）”，你可以把 `sigset_t` 想成这样：

```
第1位 → SIGHUP
第2位 → SIGINT
第3位 → SIGQUIT
...
```

每一位代表一个信号：

- 0 → 不在集合中
- 1 → 在集合中

既然要用到信号集，首先需要创建，并添加相关信号：

```c
#include <signal.h>

sigset_t set;                 // 定义一个信号集变量

sigemptyset(&set);            // 清空集合，初始化为不包含任何信号
sigfillset(&set);             // 将所有信号加入集合
sigaddset(&set, SIGINT);      // 将 SIGINT 加入集合
sigdelset(&set, SIGINT);      // 从集合中删除 SIGINT

if (sigismember(&set, SIGINT)) {
    // SIGINT 在集合中
} else {
    // SIGINT 不在集合中
}
```

这个位图的作用是：是根据具体的功能来定的，具体的功能有以下几种：

1. 信号屏蔽

   使用 `sigprocmask()` 设置进程的信号屏蔽字。

   ```c
   sigset_t newmask, oldmask;
   
   sigemptyset(&newmask);
   sigaddset(&newmask, SIGINT);   // 要阻塞的信号：SIGINT
   sigaddset(&newmask, SIGTERM);  // 再加上 SIGTERM
   
   // 阻塞 newmask 中的信号，并将原来的屏蔽字保存到 oldmask
   if (sigprocmask(SIG_BLOCK, &newmask, &oldmask) == -1) {
       perror("sigprocmask");
       exit(1);
   }
   ```

   - `SIG_BLOCK`表示将 newmask 中的信号添加到当前屏蔽字中（加进去）；
   - `SIG_UNBLOCK` 表示从当前屏蔽字中移除 newmask 中的信号（移除）；
   - `SIG_SETMASK` 表示直接将当前屏蔽字设置为 newmask（直接覆盖）。

2. 获取挂起信号

   使用 `sigpending()` 获取当前被阻塞且未处理的挂起信号集。

   ```c
   #include <signal.h>
   
   sigset_t pending;
   
   if (sigpending(&pending) == -1) {
       perror("sigpending");
       exit(1);
   }
   
   if (sigismember(&pending, SIGINT)) {
       printf("SIGINT is pending\n");
   }
   ```

3. 统一处理信号

   使用 `sigaction()` 注册信号处理函数时，通过 `sa_mask` 指定在信号处理函数执行期间额外阻塞的信号集。

   换人话就是：**sa_mask指定的信号集中的信号是在执行sigaction的handler期间额外需要被阻塞的**，如果不设置`sa_mask`，**那么系统只会屏蔽当前信号（为了避免处理函数再次被该信号中断）**。

## 案例

程序捕获 `SIGINT` 信号（通常由 `Ctrl+C` 产生），并执行自定义的处理函数，然后继续等待下一个信号。程序会一直运行，直到用户输入 `Ctrl+\`（发送 `SIGQUIT`）结束。

```c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>
#include <stdlib.h>

void sigint_handler(int sig)
{
    printf("捕获到SIGINT信号: %d\n", sig);
}

int main(void)
{
    struct sigaction new_act, old_act;

    /*设置新的处理函数*/
    new_act.sa_handler = &sigint_handler;
    sigemptyset(&new_act.sa_mask); // 除了当前信号，不屏蔽其他信号
    new_act.sa_flags = 0;         // 无特殊标志，使用默认的sa_handler函数
    
    /*注册新的函数处理方式*/
    if (sigaction(SIGINT, &new_act, &old_act) == -1)
    {
        perror("register SIGINT handler error");
        exit(1);
    }

    while(1) {

    }
}
```

## SIGCHLD信号

当子进程发生一些事情的时候，内核向父进程发送`SIGCHLD`信号：

- 子进程终止（`exit()` 或主函数返回）
- 子进程被停止（收到 `SIGSTOP`、`SIGTSTP` 等）
- 子进程从停止状态继续（收到 `SIGCONT`）

父进程都到该信号后的默认处理是忽略的。
