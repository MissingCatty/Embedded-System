# 点一个LED

```c
int main(void)
{	
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA,ENABLE);	//GPIOA时钟使能
	GPIO_InitTypeDef GPIO_InitStructure;					//配置GPIO
	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_0;				
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
	GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;		//推挽输出
	GPIO_Init(GPIOA,&GPIO_InitStructure);
	
	int tag=0;
	while(1)
	{
		if(!tag)
			GPIO_SetBits(GPIOA,GPIO_Pin_0);
		else
			GPIO_ResetBits(GPIOA,GPIO_Pin_0);
		Delay_s(1);
		tag=(tag+1)%2;
	}
}
```

如果将输出模式改为**开漏输出**，那么输出的为1时，无驱动能力，此时将LED长引脚（正极）接PA0，会发现始终不亮，因为**PA0无法输出高电平**。

# 循环点8个LED

```c
int main(void)
{	
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA,ENABLE);	//GPIOA时钟使能
	GPIO_InitTypeDef GPIO_InitStructure;					//配置GPIO
	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_All;		//GPIO_Pin_0 |	GPIO_Pin_1 | ...
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
	GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;		//推挽输出
	GPIO_Init(GPIOA,&GPIO_InitStructure);
	
	int PAx = 0x0001;
	while(1)
	{
		GPIO_Write(GPIOA,~PAx);
		Delay_s(1);
		if(PAx==0x0080)
			PAx=0x0001;
		else
			PAx <<= 1;
	}
}
```

- [15~18行]：逻辑左移，指定哪个口输出1

# 蜂鸣器

```c
int main(void)
{	
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOA,ENABLE);	//GPIOA时钟使能
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB,ENABLE);	//GPIOB时钟使能
	GPIO_InitTypeDef GPIO_InitStructure;					//配置GPIO
	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_All;		//GPIO_Pin_0 |	GPIO_Pin_1 | ...
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
	GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;		//推挽输出
	GPIO_Init(GPIOA,&GPIO_InitStructure);
	GPIO_Init(GPIOB,&GPIO_InitStructure);
	
	while(1)
	{	
		GPIO_ResetBits(GPIOB,GPIO_Pin_12);
		Delay_ms(100);
		GPIO_SetBits(GPIOB,GPIO_Pin_12);
		Delay_ms(100);
		GPIO_ResetBits(GPIOB,GPIO_Pin_12);
		Delay_ms(100);
		GPIO_SetBits(GPIOB,GPIO_Pin_12);
		Delay_ms(700);
		
	}
}
```
