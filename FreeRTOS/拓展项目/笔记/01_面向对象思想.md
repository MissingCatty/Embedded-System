我要写一个点灯程序，来控制三个LED灯的亮灭，为了方便代码的修改和管理，统一使用宏的方式定义每个LED对应的端口和引脚，以及其亮灭函数：

```c
/*
    LED0: PE5
    LED1: PE6
    LED2: PC13
*/
#define LED0_RCC   RCC_AHB1Periph_GPIOE
#define LED0_PORT  GPIOE
#define LED0_PIN   GPIO_Pin_5
#define LED0_ON()  GPIO_WriteBit(LED0_PORT, LED0_PIN, Bit_RESET)
#define LED0_OFF() GPIO_WriteBit(LED0_PORT, LED0_PIN, Bit_SET)

#define LED1_RCC   RCC_AHB1Periph_GPIOE
#define LED1_PORT  GPIOE
#define LED1_PIN   GPIO_Pin_6
#define LED1_ON()  GPIO_WriteBit(LED1_PORT, LED1_PIN, Bit_RESET)
#define LED1_OFF() GPIO_WriteBit(LED1_PORT, LED1_PIN, Bit_SET)

#define LED2_RCC   RCC_AHB1Periph_GPIOC
#define LED2_PORT  GPIOC
#define LED2_PIN   GPIO_Pin_13
#define LED2_ON()  GPIO_WriteBit(LED2_PORT, LED2_PIN, Bit_RESET)
#define LED2_OFF() GPIO_WriteBit(LED2_PORT, LED2_PIN, Bit_SET)
```

但假如说要控制10个不同的LED，那代码写起来就十分繁琐，此时就需要用面向对象的方式来统一管理。

首先，每个LED都拥有同样的属性，即：GPIOx，GPIO_Pin，RCC，打开/关闭时GPIO高低电平。

所以可以用一个结构体来管理：

```c
typedef struct {
    uint32_t      rcc;
    uint16_t      pin;
    GPIO_TypeDef *port;
    BitAction     on_level;
    BitAction     off_level;
} led_conf_t; // _t表示是一个typedef过来的东西
```

使用时，可以实例化上述结构体类型：

```c
led_conf_t led0 = {
    .rcc = RCC_AHB1Periph_GPIOE,
    .port       = GPIOE,
    .pin        = GPIO_Pin_5,
    .on_level   = Bit_RESET,
    .off_level  = Bit_SET,
};

led_conf_t led1 = {
    .rcc = RCC_AHB1Periph_GPIOE,
    .port       = GPIOE,
    .pin        = GPIO_Pin_6,
    .on_level   = Bit_RESET,
    .off_level  = Bit_SET
};

led_conf_t led2 = {
    .rcc = RCC_AHB1Periph_GPIOC,
    .port       = GPIOC,
    .pin        = GPIO_Pin_13,
    .on_level   = Bit_RESET,
    .off_level  = Bit_SET
};
```

用一个初始化函数，统一初始化：

```c
void LED_Init(const led_conf_t * led)
{
    RCC_AHB1PeriphClockCmd(led->rcc, ENABLE);

    GPIO_InitTypeDef GPIO_InitStructure;
    GPIO_StructInit(&GPIO_InitStructure);
    GPIO_InitStructure.GPIO_Mode  = GPIO_Mode_OUT;
    GPIO_InitStructure.GPIO_OType = GPIO_OType_PP;
    GPIO_InitStructure.GPIO_Pin   = led->pin;
    GPIO_InitStructure.GPIO_Speed = GPIO_Medium_Speed;
    GPIO_Init(led->port, &GPIO_InitStructure);

    GPIO_WriteBit(led->port, led->pin, led->off_level);
}
```

之后，在`main.c`文件中初始化用到的led：

```c
int main(void)
{
    led_conf_t led0 = {
        .rcc = RCC_AHB1Periph_GPIOE,
        .port       = GPIOE,
        .pin        = GPIO_Pin_5,
        .on_level   = Bit_RESET,
        .off_level  = Bit_SET,
    };

    led_conf_t led1 = {
        .rcc = RCC_AHB1Periph_GPIOE,
        .port       = GPIOE,
        .pin        = GPIO_Pin_6,
        .on_level   = Bit_RESET,
        .off_level  = Bit_SET
    };

    led_conf_t led2 = {
        .rcc = RCC_AHB1Periph_GPIOC,
        .port       = GPIOC,
        .pin        = GPIO_Pin_13,
        .on_level   = Bit_RESET,
        .off_level  = Bit_SET
    };

    LED_Init(&led0);
    LED_Init(&led1);
    LED_Init(&led2);

    while (1)
    {
        LED_ON(&led0);
        LED_ON(&led1);
        LED_ON(&led2);
        LED_Delay();

        LED_OFF(&led0);
        LED_OFF(&led1);
        LED_OFF(&led2);
        LED_Delay();
    }
}
```

现在又有一个问题，如果把开发板上所有的要用到的设备全部在`main.c`文件里声明和构造（如上面的led0/1/2），看起来就非常繁杂。`main.c`文件里最好只留各个器件的初始化语句和功能执行语句。

所以，就要把声明器件的语句统一放到另一个文件进行管理，这里新建一个`board.c`和`board.h`文件，用于存放开发板上使用到的设备的构造语句。

**board.c**

```c
#include "board.h"

led_conf_t led0 = {
    .rcc = RCC_AHB1Periph_GPIOE,
    .port       = GPIOE,
    .pin        = GPIO_Pin_5,
    .on_level   = Bit_RESET,
    .off_level  = Bit_SET,
};

led_conf_t led1 = {
    .rcc = RCC_AHB1Periph_GPIOE,
    .port       = GPIOE,
    .pin        = GPIO_Pin_6,
    .on_level   = Bit_RESET,
    .off_level  = Bit_SET
};

led_conf_t led2 = {
    .rcc = RCC_AHB1Periph_GPIOC,
    .port       = GPIOC,
    .pin        = GPIO_Pin_13,
    .on_level   = Bit_RESET,
    .off_level  = Bit_SET
};
```

**board.h**

```c
#ifndef BOARD_H
#define BOARD_H

#include "LED.h"

extern led_conf_t led0;
extern led_conf_t led1;
extern led_conf_t led2;

#endif
```

之后在`main.h`里包含所有要引用到的头文件：

```c
#include "LED.h"
#include "board.h"
```
